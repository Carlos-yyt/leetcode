```python
class Solution:
    def maximumUnits(self, boxTypes: List[List[int]], truckSize: int) -> int:
        ans=0
        boxTypes=sorted(boxTypes, key=lambda boxTypes: boxTypes[1],reverse = True)
        for boxes in boxTypes:
            if truckSize==0:
                break
            elif truckSize<boxes[0]:
                ans+=truckSize*boxes[1]
                break
            else:
                ans+=boxes[0]*boxes[1]
                truckSize-=boxes[0]

        return ans
```
